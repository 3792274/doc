tar zxf apache-maven-3.3.9-bin.tar.gz
mv apache-maven-3.3.9 /usr/local/maven3

vim /etc/profile
#在适当的位置添加
export M2_HOME=/usr/local/maven3
export JAVA_HOME=/usr/lib/jvm/java-8-openjdk-amd64
export PATH=$PATH:$JAVA_HOME/bin:$M2_HOME/bin

source /etc/profile
